        .box {
            background-color: rgb(83 134 234);

            display: flow-root;
        }

        .p_left {
            float: left;
            width: 10%;
            height: 100px;
            background-color: white;
            border: 1px solid black;
            padding: 1pt;
        }
        .p_middle {
            float: left;
            width: 20%;
            height: 100px;
            background-color: white;
            border: 1px solid black;
            padding: 1pt;
        }
        .p_rigth {
        /* width уже не важен, займёт всё пространство справа */

            padding: 10px;
        }
    </style>
</head>
